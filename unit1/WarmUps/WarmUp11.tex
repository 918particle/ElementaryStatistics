\section{Formual Area}

\begin{enumerate}
\item \textit{Mean of a uniform distribution between $a$ and $b$: $\mu = (b+a)/2$.}
\item \textit{Variance of a uniform distribution between $a$ and $b$: $\sigma^2 = (b-a)^2/12$.}
\end{enumerate}
